<script>
let basket = document.getElementById("basket");
let scoreDisplay = document.getElementById("score");
let score = 0;

// Bewegung Maus / Touch
document.addEventListener("mousemove", e => { 
  basket.style.left = Math.min(Math.max(e.clientX-40,0), window.innerWidth-80) + "px"; 
});
document.addEventListener("touchmove", e => { 
  basket.style.left = Math.min(Math.max(e.touches[0].clientX-40,0), window.innerWidth-80) + "px"; 
});

// Items erstellen
function createItem(){
  let item = document.createElement("div");
  item.classList.add("item");
  let icons = ["üóëÔ∏è","üß¥","üì¶","üçæ","üßÉ"];
  item.textContent = icons[Math.floor(Math.random()*icons.length)];
  item.style.left = Math.random()*(window.innerWidth-40)+"px";
  document.getElementById("game").appendChild(item);
  let fallSpeed = 3 + Math.random()*3;

  function fall(){
    let top = item.offsetTop + fallSpeed;
    item.style.top = top + "px";
    let b = basket.getBoundingClientRect();
    let i = item.getBoundingClientRect();
    if(i.bottom>=b.top && i.top<=b.bottom && i.right>=b.left && i.left<=b.right){
      score++;
      scoreDisplay.textContent = "Score: " + score;
      item.remove();
      return;
    }
    if(top>window.innerHeight){ item.remove(); return; }
    requestAnimationFrame(fall);
  }
  fall();
}
setInterval(createItem,1000);
